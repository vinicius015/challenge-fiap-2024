CREATE TABLE EURO_DEPARTAMENT (
    ID BIGINT PRIMARY KEY,
    T_DEPARTAMENT VARCHAR(256) NOT NULL,
    T_DESCRIPTION VARCHAR(MAX) NULL
);

CREATE TABLE EURO_POSITION (
    ID BIGINT PRIMARY KEY,
    T_POSITION VARCHAR(256) NOT NULL,
    EURO_DEPARTAMENT_ID BIGINT NOT NULL,
    CONSTRAINT EURO_POSITION_FK FOREIGN KEY (EURO_DEPARTAMENT_ID) REFERENCES EURO_DEPARTAMENT(ID)
);

CREATE TABLE EURO_COMPLIANCE_TITLE (
    ID BIGINT PRIMARY KEY,
    T_TITLE VARCHAR(256) NOT NULL,
    T_CONTENT VARCHAR(MAX) NULL,
    N_VERSION NUMERIC(10) NULL
);

CREATE TABLE EURO_COMPLIANCE_SUBTITLE (
    ID BIGINT PRIMARY KEY,
    T_SUBTITLE VARCHAR(256) NOT NULL,
    T_CONTENT VARCHAR(MAX) NULL,
    EURO_COMPLIANCE_TITLE_ID BIGINT NOT NULL,
    CONSTRAINT EURO_COMPLIANCE_SUBTITLE_FK FOREIGN KEY (EURO_COMPLIANCE_TITLE_ID) REFERENCES EURO_COMPLIANCE_TITLE(ID)
);

CREATE TABLE EURO_USER (
    ID BIGINT PRIMARY KEY,
    T_USERNAME VARCHAR(50) NOT NULL UNIQUE,
    T_NAME VARCHAR(256) NOT NULL,
    T_EMAIL VARCHAR(256) NOT NULL,
    T_CPF VARCHAR(14) NOT NULL,
    T_NUMBER NUMERIC(13) NULL,
    EURO_POSITION_ID BIGINT NOT NULL,
    CONSTRAINT EURO_USER_FK FOREIGN KEY (EURO_POSITION_ID) REFERENCES EURO_POSITION(ID)
);

CREATE TABLE EURO_AUTH (
    ID BIGINT PRIMARY KEY,
    T_PASS VARCHAR(MAX) NOT NULL,
    B_ACTIVED BOOLEAN NOT NULL,
    DT_LAST_AUTH DATETIME NULL,
    DT_GENERATED DATETIME NOT NULL,
    ID1 BIGINT NULL
);

CREATE TABLE EURO_COMPLIANCE_MANUAL (
    EURO_USER_ID BIGINT NOT NULL,
    EURO_COMPLIANCE_TITLE_ID BIGINT NOT NULL,
    PRIMARY KEY (EURO_USER_ID, EURO_COMPLIANCE_TITLE_ID),
    CONSTRAINT EURO_COMPLIANCE_MANUAL_USER_FK FOREIGN KEY (EURO_USER_ID) REFERENCES EURO_USER(ID),
    CONSTRAINT EURO_COMPLIANCE_MANUAL_TITLE_FK FOREIGN KEY (EURO_COMPLIANCE_TITLE_ID) REFERENCES EURO_COMPLIANCE_TITLE(ID)
);
